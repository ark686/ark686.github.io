2.1 Basic Description Statistics ​

python
import numpy as np
import pandas as pd

# Sample data: starting salaries ( thousands ) for ECE graduates
salaries = np.array([58 , 62, 56, 67, 66, 58, 60, 61, 70, 61,
    65, 60, 61, 80, 62, 75, 60, 68, 57, 65])

# Measures of center
mean_salary = np.mean( salaries ) # 63.6
median_salary = np.median( salaries ) # 61.5

# Measures of variability
variance = np.var(salaries , ddof =1) # Sample variance (ddof =1)
std_dev = np.std(salaries , ddof =1) # Sample std deviation
range_val = np.ptp( salaries ) # Peak -to -peak (range)

# Quartiles and percentiles
q1 = np.percentile(salaries , 25)
q3 = np.percentile(salaries , 75)
iqr = q3 - q1

print(f"Mean: ${ mean_salary :.1f}k")
print(f"Median : ${ median_salary :.1f}k")
print(f"Std Dev: ${ std_dev :.2f}k")
print(f"IQR: ${iqr :.1f}k")

# Using Pandas for comprehensive summary
df = pd.DataFrame({'Salary ': salaries })
print(df.describe ())

2.3 Stem-and-Leaf Display ​

python
def stem_leaf_display (data, scale_factor=10):
    """Create a simple stem -and -leaf display ."""
    sorted_data = sorted (data)
    stems = {}

    for value in sorted_data:
        stem = int(value // scale_factor)
        leaf = int(value % scale_factor)
        if stem not in stems:
            stems[stem] = []
        stems[stem].append(leaf)

    print("Stem | Leaves")
    print("-" * 20)
    for stem in sorted (stems.keys()):
        leaves = ' '.join(map(str , stems[stem ]))
        print(f" {stem} | {leaves }")

# Apply to salary data
stem_leaf_display (salaries, scale_factor=10)

2.4 Analyzing Network Latency Data ​

python
import numpy as np
import pandas as pd
import matplotlib .pyplot as plt
from scipy import stats

# Simulated network latency data ( milliseconds )
np.random.seed (42)
# Mixture : mostly normal with some high - latency outliers
normal_latency = np.random.normal(loc =50, scale =10, size =90)
high_latency = np.random.exponential (scale =30, size =10) + 80
latency_data = np.concatenate([ normal_latency , high_latency ])

# Comprehensive descriptive analysis
def describe_data (data , name="Data"):
    """Print comprehensive descriptive statistics ."""
    print(f"=== {name} ===")
    print(f"Sample size: n = {len(data)}")
    print(f"\nMeasures of Center :")
    print(f" Mean: {np.mean(data):.2f} ms")
    print(f" Median : {np.median (data):.2f} ms")
    print(f" Mode: {stats.mode(data.round ())[0]:.0f} ms")

    print(f"\nMeasures of Variability:")
    print(f" Range: {np.ptp(data):.2f} ms")
    print(f" Variance : {np.var(data , ddof =1) :.2f} ms^2")
    print(f" Std Dev: {np.std(data , ddof =1) :.2f} ms")
    print(f" IQR: {np.percentile(data , 75) - np.percentile(data, 25):.2f} ms")
    print(f" CV: {100* np.std(data , ddof =1)/np.mean(data):.1f}%")

    print(f"\nFive-Number Summary:")
    q1 , median , q3 = np. percentile (data , [25, 50, 75])
    print(f" Min: {np.min(data):.2f}")
    print(f" Q1: {q1 :.2f}")
    print(f" Q2: {median :.2f}")
    print(f" Q3: {q3 :.2f}")
    print(f" Max: {np.max(data):.2f}")

    # Detect outliers using IQR rule
    iqr = q3 - q1
    lower_fence = q1 - 1.5 * iqr
    upper_fence = q3 + 1.5 * iqr
    outliers = data [( data < lower_fence ) | (data > upper_fence )]
    print(f"\nOutliers (IQR rule): {len( outliers )} detected")
    if len( outliers ) > 0:
        poutliers = sorted(outliers)
        for x in poutliers:
            print(f"  {x:.2f}")

describe_data(latency_data, "Network Latency Analysis")

2.5 Comparative Analysis with Seaborn ​

python
import seaborn as sns

# Compare latency across different server types
np.random.seed (123)
data = {
    'Server': ['A']*30 + ['B']*30 + ['C']*30 ,
    'Latency': np. concatenate ([
        np.random.normal (45, 8, 30) , # Server A: low latency
        np.random.normal (60, 15, 30) , # Server B: higher , more variable
        np.random. exponential (20, 30) + 40 # Server C: right -skewed
    ])
}
df = pd.DataFrame(data)

# Side-by-side box plots
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

# Box plot
sns.boxplot(x='Server', y='Latency', data=df, ax=axes[0])
axes[0].set_title('Latency Comparison by Server')
axes[0].set_ylabel('Latency (ms)')

# Violin plot (shows full distribution )
sns.violinplot(x='Server', y='Latency', data=df, ax=axes[1])
axes[1].set_title('Latency Distribution by Server')
axes[1].set_ylabel('Latency (ms)')

plt.tight_layout()
plt.show()

# Summary statistics by group
print(df.groupby('Server')['Latency'].describe())
