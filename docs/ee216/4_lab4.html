Lab 4 - Conditional Probability and Independence ​

1. Overview ​

Conditional Probability:

  • Answers the question: “How does the probability of event A change when we know that event B has occurred?”

Independence:

  • Captures situations where knowing about one event provides no information about another: “The occurrence of B does not affect the probability of A.”

2. Example - Implementing Bayes' Theorem ​

python
import numpy as np

def bayes_theorem(prior , likelihood , evidence=None):
    """
    Compute posterior probability using Bayes' theorem.
    Parameters:
    -----------
    prior : float or array
        Prior probability P(A) or array of priors [P(A1), P(A2), ...]
    likelihood : float or array
        Likelihood P(B|A) or array [P(B|A1), P(B|A2), ...]
    evidence : float , optional
        P(B). If None , computed from prior and likelihood.

    Returns:
    --------
    posterior : float or array
        Posterior probability P(A|B)
    """
    prior = np.array(prior)
    likelihood = np.array(likelihood)

    # Compute evidence using law of total probability if not given
    if evidence is None:
        evidence = np.sum(prior * likelihood)

    posterior = (likelihood * prior) / evidence
    return posterior

# Medical diagnosis example
prior_disease = 0.001 # P(D)
sensitivity = 0.99 # P(T+|D)
false_positive = 0.02 # P(T+|D')

# Compute P(D|T+)
posterior = bayes_theorem(
    prior=[prior_disease , 1-prior_disease],
    likelihood=[sensitivity , false_positive]
)
print(f"P(Disease | Positive Test) = {posterior [0]:.4f}")
# Output: P(Disease | Positive Test) = 0.0472

3. Example - Monte Carlo Estimation of Conditional Probability ​

python
import numpy as np

def estimate_conditional_prob(n_simulations =100000):
    """
    Estimate P(sum >= 9 | at least one die shows 6)
    using Monte Carlo simulation.
    """
    np.random.seed(42)

    # Roll two dice
    die1 = np.random.randint(1, 7, n_simulations)
    die2 = np.random.randint(1, 7, n_simulations)

    # Events
    sum_dice = die1 + die2
    at_least_one_six = (die1 == 6) | (die2 == 6)
    sum_geq_9 = sum_dice >= 9

    # Count outcomes where B occurs (at least one 6)
    n_B = np.sum(at_least_one_six)

    # Count outcomes where both A and B occur
    n_A_and_B = np.sum(at_least_one_six & sum_geq_9)

    # Conditional probability P(A|B)
    conditional_prob = n_A_and_B / n_B

    print(f"Simulated P(sum >= 9 | at least one 6) = {conditional_prob :.4f}")

    # Exact calculation
    # B = at least one 6: 11 outcomes out of 36
    # A B = {(6,3), (3,6), (6,4), (4,6), (6,5), (5,6), (6,6)}: 7 outcomes
    exact = 7/11
    print(f"Exact P(sum >= 9 | at least one 6) = {exact :.4f}")

estimate_conditional_prob()

4. Example - Reliability Analysis ​

python
import numpy as np

def series_reliability(component_reliabilities):
    """Compute reliability of a series system."""
    return np.prod(component_reliabilities)

def parallel_reliability(component_reliabilities):
    """Compute reliability of a parallel system."""
    return 1 - np.prod(1 - np.array(component_reliabilities))

def series_parallel_reliability(subsystems):
    """
    Compute reliability of series -parallel system.

    Parameters:
    -----------
    subsystems : list of lists
        Each inner list contains reliabilities of components
        in a parallel subsystem. Subsystems are in series.
    """
    subsystem_reliabilities = [parallel_reliability(sub) for sub in subsystems]
    return series_reliability(subsystem_reliabilities)

# Example: Two parallel subsystems in series
# Subsystem 1: Components with R=0.9
# Subsystem 2: Components with R=0.85
subsystem1 = [0.9, 0.9] # Two parallel components
subsystem2 = [0.85, 0.85, 0.85] # Three parallel components

R1 = parallel_reliability(subsystem1)
R2 = parallel_reliability(subsystem2)
R_total = series_parallel_reliability([subsystem1 , subsystem2])

print(f"Subsystem 1 reliability: {R1:.4f}")
print(f"Subsystem 2 reliability: {R2:.4f}")
print(f"System reliability: {R_total :.4f}")
# Output:
# Subsystem 1 reliability: 0.9900
# Subsystem 2 reliability: 0.9966
# System reliability: 0.9867

5. Example - Visualizing Prior to Posterior Update ​

python
"""
Bayesian Analysis of Medical Test Results

This program demonstrates Bayesian inference for a medical test scenario:
- Rare condition affecting 0.1% of population
- Test with 95% accuracy (sensitivity and specificity)
- Analysis of prior, likelihood, and posterior probabilities

Author: Generated for educational purposes
"""

import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from matplotlib.patches import Rectangle
import pandas as pd

# Set up the plotting style
plt.style.use('seaborn-v0_8')
sns.set_palette("husl")

class BayesianMedicalTest:
    def __init__(self, disease_prevalence=0.01, test_sensitivity=0.95, test_specificity=0.95):
        """
        Initialize the Bayesian medical test analysis
        
        Parameters:
        - disease_prevalence: P(Disease) - prior probability
        - test_sensitivity: P(Test+|Disease) - true positive rate
        - test_specificity: P(Test-|No Disease) - true negative rate
        """
        self.prior_disease = disease_prevalence
        self.prior_no_disease = 1 - disease_prevalence
        self.sensitivity = test_sensitivity
        self.specificity = test_specificity
        self.false_positive_rate = 1 - test_specificity
        self.false_negative_rate = 1 - test_sensitivity
        
        # Calculate marginal probability of positive test
        self.p_test_positive = (self.sensitivity * self.prior_disease + 
                               self.false_positive_rate * self.prior_no_disease)
        
        # Calculate posterior probabilities using Bayes' theorem
        self.posterior_disease_given_positive = (self.sensitivity * self.prior_disease) / self.p_test_positive
        self.posterior_no_disease_given_positive = (self.false_positive_rate * self.prior_no_disease) / self.p_test_positive
        
    def print_analysis(self):
        """Print detailed analysis of the probabilities"""
        print("=" * 60)
        print("BAYESIAN MEDICAL TEST ANALYSIS")
        print("=" * 60)
        print(f"Disease prevalence (Prior): {self.prior_disease:.1%}")
        print(f"Test sensitivity: {self.sensitivity:.1%}")
        print(f"Test specificity: {self.specificity:.1%}")
        print("-" * 60)
        print("LIKELIHOOD PROBABILITIES:")
        print(f"P(Test+|Disease): {self.sensitivity:.1%}")
        print(f"P(Test+|No Disease): {self.false_positive_rate:.1%}")
        print("-" * 60)
        print("MARGINAL PROBABILITY:")
        print(f"P(Test+): {self.p_test_positive:.3f} or {self.p_test_positive:.1%}")
        print("-" * 60)
        print("POSTERIOR PROBABILITIES (Given positive test):")
        print(f"P(Disease|Test+): {self.posterior_disease_given_positive:.3f} or {self.posterior_disease_given_positive:.1%}")
        print(f"P(No Disease|Test+): {self.posterior_no_disease_given_positive:.3f} or {self.posterior_no_disease_given_positive:.1%}")
        print("=" * 60)
        
    def plot_comprehensive_analysis(self):
        """Create comprehensive visualization of the Bayesian analysis"""
        fig = plt.figure(figsize=(16, 12))
        
        # 1. Prior Probabilities (Top Left)
        ax1 = plt.subplot(2, 3, 1)
        categories = ['Disease', 'No Disease']
        prior_probs = [self.prior_disease, self.prior_no_disease]
        colors = ['#FF6B6B', '#4ECDC4']
        
        bars1 = ax1.bar(categories, prior_probs, color=colors, alpha=0.8, edgecolor='black', linewidth=2)
        ax1.set_title('Prior Probabilities\n(Population Base Rates)', fontsize=14, fontweight='bold')
        ax1.set_ylabel('Probability', fontsize=12)
        ax1.set_ylim(0, 1)
        
        # Add percentage labels on bars
        for bar, prob in zip(bars1, prior_probs):
            height = bar.get_height()
            ax1.text(bar.get_x() + bar.get_width()/2., height + 0.01,
                    f'{prob:.1%}', ha='center', va='bottom', fontweight='bold')
        
        # 2. Likelihood Probabilities (Top Middle)
        ax2 = plt.subplot(2, 3, 2)
        likelihood_categories = ['P(Test+|Disease)', 'P(Test+|No Disease)']
        likelihood_probs = [self.sensitivity, self.false_positive_rate]
        
        bars2 = ax2.bar(likelihood_categories, likelihood_probs, color=['#FF9F43', '#FF6B6B'], 
                       alpha=0.8, edgecolor='black', linewidth=2)
        ax2.set_title('Likelihood Probabilities\n(Test Accuracy)', fontsize=14, fontweight='bold')
        ax2.set_ylabel('Probability', fontsize=12)
        ax2.set_ylim(0, 1)
        ax2.tick_params(axis='x', rotation=15)
        
        for bar, prob in zip(bars2, likelihood_probs):
            height = bar.get_height()
            ax2.text(bar.get_x() + bar.get_width()/2., height + 0.01,
                    f'{prob:.1%}', ha='center', va='bottom', fontweight='bold')
        
        # 3. Posterior Probabilities (Top Right)
        ax3 = plt.subplot(2, 3, 3)
        posterior_categories = ['Disease|Test+', 'No Disease|Test+']
        posterior_probs = [self.posterior_disease_given_positive, self.posterior_no_disease_given_positive]
        
        bars3 = ax3.bar(posterior_categories, posterior_probs, color=['#A8E6CF', '#FFB6C1'], 
                       alpha=0.8, edgecolor='black', linewidth=2)
        ax3.set_title('Posterior Probabilities\n(After Positive Test)', fontsize=14, fontweight='bold')
        ax3.set_ylabel('Probability', fontsize=12)
        ax3.set_ylim(0, 1)
        ax3.tick_params(axis='x', rotation=15)
        
        for bar, prob in zip(bars3, posterior_probs):
            height = bar.get_height()
            ax3.text(bar.get_x() + bar.get_width()/2., height + 0.01,
                    f'{prob:.1%}', ha='center', va='bottom', fontweight='bold')
        
        # 4. Bayes' Theorem Visualization (Bottom Left)
        ax4 = plt.subplot(2, 3, 4)
        ax4.text(0.5, 0.8, "Bayes' Theorem", ha='center', va='center', 
                fontsize=16, fontweight='bold', transform=ax4.transAxes)
        
        formula = (f"P(Disease|Test+) = \n\n"
                  f"P(Test+|Disease) × P(Disease)\n"
                  f"─────────────────────────────\n"
                  f"         P(Test+)\n\n"
                  f"= {self.sensitivity:.2f} × {self.prior_disease:.2f}\n"
                  f"  ─────────────────\n"
                  f"     {self.p_test_positive:.3f}\n\n"
                  f"= {self.posterior_disease_given_positive:.3f}")
        
        ax4.text(0.5, 0.4, formula, ha='center', va='center', 
                fontsize=11, transform=ax4.transAxes, 
                bbox=dict(boxstyle="round,pad=0.3", facecolor="lightblue", alpha=0.5))
        ax4.set_xlim(0, 1)
        ax4.set_ylim(0, 1)
        ax4.axis('off')
        
        # 5. Population Visualization (Bottom Middle and Right)
        ax5 = plt.subplot(2, 3, (5, 6))
        
        # Create a population grid visualization
        population_size = 1000
        diseased = int(population_size * self.prior_disease)
        healthy = population_size - diseased
        
        # Test results
        true_positives = int(diseased * self.sensitivity)
        false_negatives = diseased - true_positives
        false_positives = int(healthy * self.false_positive_rate)
        true_negatives = healthy - false_positives
        
        # Create pie chart showing test results
        labels = ['True Positives', 'False Positives', 'True Negatives', 'False Negatives']
        sizes = [true_positives, false_positives, true_negatives, false_negatives]
        colors_pie = ['#2ECC71', '#E74C3C', '#3498DB', '#F39C12']
        explode = (0.05, 0.05, 0, 0)  # explode positive results
        
        wedges, texts, autotexts = ax5.pie(sizes, labels=labels, colors=colors_pie, 
                                          autopct='%1.0f', startangle=90, explode=explode)
        ax5.set_title(f'Test Results in Population of {population_size}\n'
                     f'Positive Predictive Value: {self.posterior_disease_given_positive:.1%}', 
                     fontsize=14, fontweight='bold')
        
        # Add text box with interpretation
        interpretation = (f"Out of {true_positives + false_positives} positive tests:\n"
                         f"• {true_positives} are true positives (have disease)\n"
                         f"• {false_positives} are false positives (don't have disease)\n\n"
                         f"Therefore, only {self.posterior_disease_given_positive:.1%} of positive\n"
                         f"tests indicate the person actually has the disease!")
        
        plt.figtext(0.02, 0.02, interpretation, fontsize=10, 
                   bbox=dict(boxstyle="round,pad=0.5", facecolor="lightyellow", alpha=0.8))
        
        plt.tight_layout()
        plt.show()
        
    def plot_sensitivity_analysis(self):
        """Plot how posterior probability changes with different test accuracies"""
        accuracies = np.linspace(0.5, 0.99, 50)
        posteriors = []
        
        for acc in accuracies:
            p_test_pos = acc * self.prior_disease + (1-acc) * self.prior_no_disease
            posterior = (acc * self.prior_disease) / p_test_pos
            posteriors.append(posterior)
        
        plt.figure(figsize=(10, 6))
        plt.plot(accuracies, posteriors, 'b-', linewidth=3, label='Posterior P(Disease|Test+)')
        plt.axhline(y=self.posterior_disease_given_positive, color='r', linestyle='--', 
                   linewidth=2, label=f'Current test (95% acc): {self.posterior_disease_given_positive:.1%}')
        plt.axvline(x=0.95, color='r', linestyle='--', alpha=0.7)
        
        plt.xlabel('Test Accuracy (Sensitivity = Specificity)', fontsize=12)
        plt.ylabel('Posterior Probability P(Disease|Test+)', fontsize=12)
        plt.title('How Test Accuracy Affects Posterior Probability\n(for 1% disease prevalence)', 
                 fontsize=14, fontweight='bold')
        plt.grid(True, alpha=0.3)
        plt.legend()
        plt.ylim(0, 1)
        plt.show()

def main():
    """Main function to run the Bayesian analysis"""
    print("Bayesian Medical Test Analysis")
    print("=" * 50)
    
    # Create the analysis
    test = BayesianMedicalTest(disease_prevalence=0.01, test_sensitivity=0.95, test_specificity=0.95)
    
    # Print numerical analysis
    test.print_analysis()
    
    # Create visualizations
    print("\nGenerating comprehensive visualization...")
    test.plot_comprehensive_analysis()
    
    print("\nGenerating sensitivity analysis...")
    test.plot_sensitivity_analysis()
    
    # Additional insight
    print("\n" + "="*60)
    print("KEY INSIGHTS:")
    print("="*60)
    print(f"Even with a 95% accurate test, there's only a {test.posterior_disease_given_positive:.1%} chance")
    print("you actually have the disease if you test positive!")
    print("\nThis counterintuitive result occurs because:")
    print("1. The disease is rare (1% prevalence)")
    print("2. False positives outnumber true positives")
    print("3. Base rate matters enormously in medical testing")
    print("\nThis is why doctors often order confirmatory tests!")

if __name__ == "__main__":
    main()
