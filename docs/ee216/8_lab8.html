2.1.1 CI for the Mean ​

python
import numpy as np
from scipy import stats

# Sample data: Network latency measurements (ms)
latency = np.array ([23.5 , 25.1 , 24.2 , 26.3 , 22.8 , 24.9 ,
25.7 , 23.1 , 24.6 , 25.4 , 24.0 , 25.8])

# Basic statistics
n = len( latency )
xbar = np.mean( latency )
s = np.std(latency , ddof=1) # Sample std (ddof =1 for n -1)
se = s / np.sqrt(n) # Standard error

print(f"Sample size: n = {n}")
print(f"Sample mean:̄ x = {xbar :.3f} ms")
print(f"Sample std: s = {s:.3f} ms")
print(f"Std error: SE = {se :.3f} ms")

# 95% CI using t- distribution
alpha = 0.05
t_crit = stats.t.ppf (1 - alpha /2, df=n -1)
margin_error = t_crit * se
ci_lower = xbar - margin_error
ci_upper = xbar + margin_error

print(f"\n95% Confidence Interval :")
print(f"t critical value: t_{alpha /2},{n -1} = {t_crit :.3f}")
print(f"Margin of error: E = { margin_error :.3f} ms")
print(f"CI: ({ ci_lower :.3f}, { ci_upper :.3f}) ms")

# Using scipy 's built -in function
ci = stats.t. interval ( confidence =0.95 , df=n-1, loc=xbar , scale=se)
print(f"\nUsing scipy: ({ci[0]:.3f}, {ci[1]:.3f}) ms")

import numpy as np
from scipy import stats

# Defect rate example
x = 23 # Number of defectives
n = 500 # Sample size
p_hat = x / n

# Check normal approximation conditions
print(f"Check conditions : n*p_hat = {n*p_hat :.1f} >= 10: {n*p_hat >= 10}")
print(f"Check conditions : n*(1- p_hat) = {n*(1- p_hat):.1f} >= 10: {n*(1- p_hat) >= 10}")

# 95% Wald CI
alpha = 0.05
z_crit = stats.norm.ppf (1 - alpha /2)

# Wilson score interval (more accurate )
from statsmodels.stats.proportion import proportion_confint
ci_wilson = proportion_confint(x, n, alpha =0.05 , method='wilson')
print(f"95% CI(Wilson): ({ ci_wilson[0]:.4f}, { ci_wilson[1]:.4f})")

2.2 Visualizing Confidence Intervals ​

python
import numpy as np
import matplotlib .pyplot as plt
from scipy import stats

# Simulation : Generate multiple CIs and check coverage
np.random.seed(42)
mu_true = 50 # True population mean
sigma = 10 # Population std
n = 25 # Sample size
n_simulations = 100
alpha = 0.05

# Storage for CIs
cis = []
covers_mu = []

for _ in range( n_simulations ):
    # Generate sample
    sample = np.random.normal(mu_true, sigma, n)
    xbar = np.mean(sample)
    s = np.std(sample, ddof=1)
    se = s / np.sqrt(n)

    # Compute 95% CI
    t_crit = stats.t.ppf(1 - alpha/2, df=n-1)
    ci = (xbar - t_crit*se, xbar + t_crit*se)
    cis.append(ci)
    covers_mu.append(ci[0] <= mu_true <= ci[1])

# Calculate coverage rate
coverage = sum( covers_mu ) / n_simulations
print(f" Empirical coverage : { coverage *100:.1f}%")

# Plot first 50 CIs
fig , ax = plt. subplots ( figsize =(10 , 8))
for i in range (50):
    color = 'green' if covers_mu [i] else 'red'
    ax.plot ([ cis[i][0] , cis[i][1]] , [i, i], color=color , linewidth =2)
    ax.plot ([( cis[i][0] + cis[i][1]) /2], [i], 'ko', markersize =3)

ax.axvline(mu_true , color='blue', linestyle ='--', label=f' = { mu_true }')
ax.set_xlabel('Value')
ax.set_ylabel('Simulation')
ax.set_title(f'95% Confidence Intervals ( Coverage : { coverage*100:.1f}%) ')
ax.legend()
plt.tight_layout()

2.4 Implementing Hypothesis Tests ​

python
import numpy as np
from scipy import stats

# One-sample z-test (sigma known)
def z_test_one_sample (data , mu0 , sigma , alternative ='two -sided '):
    """
    Perform one -sample z-test when population sigma is known.
     Parameters :
    -----------
    data : array -like - Sample data
    mu0 : float - Hypothesized population mean
    sigma : float - Known population standard deviation
    alternative : str - 'two -sided ', 'greater ', or 'less '

    Returns :
    --------
    z_stat : float - Test statistic
    p_value : float - P-value
    """
    n = len(data)
    x_bar = np.mean(data)
    z_stat = (x_bar - mu0) / (sigma / np.sqrt(n))

    if alternative == 'two-sided':
        p_value = 2 * (1 - stats.norm.cdf(abs(z_stat)))
    elif alternative == 'greater':
        p_value = 1 - stats.norm.cdf(z_stat)
    else: # 'less '
        p_value = stats.norm.cdf(z_stat)

    return z_stat , p_value

# Example usage
np.random.seed(42)
data = np.random.normal (5.12 , 0.3, 36) # Sample data
z, p = z_test_one_sample (data , mu0 =5.0 , sigma =0.3 , alternative ='two-sided')
print(f"z-statistic : {z:.4f}")
print(f"p-value: {p:.4f}")

2.6 Power Analysis ​

2.6.1 Power and Sample Size Calculations ​

python
import numpy as np
from scipy import stats
import matplotlib .pyplot as plt

def power_z_test (mu0, mu_alt, sigma, n, alpha=0.05, alternative='greater'):
    """
    Calculate power for one-sample z-test.
    """
    se = sigma / np.sqrt(n)

    if alternative == 'greater':
        z_crit = stats.norm.ppf (1 - alpha)
        power = 1 - stats.norm.cdf(z_crit - (mu_alt - mu0) / se)
    elif alternative == 'less':
        z_crit = stats.norm.ppf(alpha)
        power = stats.norm.cdf(z_crit - (mu_alt - mu0) / se)
    else: # two -sided
        z_crit = stats.norm.ppf (1 - alpha /2)
    power = (1 - stats.norm.cdf(z_crit - (mu_alt - mu0) / se) +
        stats.norm.cdf(-z_crit - (mu_alt - mu0) / se))
    return power

def sample_size_z_test (mu0, mu_alt, sigma, alpha=0.05, power=0.80, alternative='greater'):
    """
    Calculate required sample size for z-test.
    """
    if alternative == 'greater' or alternative == 'less':
        z_alpha = stats.norm.ppf (1 - alpha)
    else:
        z_alpha = stats.norm.ppf (1 - alpha /2)

    z_beta = stats.norm.ppf(power)
    n = (( sigma * ( z_alpha + z_beta)) / (mu_alt - mu0))**2
    return int(np.ceil(n))

# Example : Power analysis
mu0 = 100
sigma = 15
alpha = 0.05

# Power for different alternatives and sample sizes
print("=== Power Analysis ===")
print(f"H0: mu = {mu0}, sigma = {sigma}, alpha = {alpha}")
print("\nPower at mu' = 105:")
for n in [25, 50, 100, 200]:
    pwr = power_z_test (mu0 , 105, sigma , n, alpha , 'greater')
    print(f"n = {n:3d}: Power = {pwr :.4f}")

# Sample size calculation
print("\n=== Sample Size Calculation ===")
required_n = sample_size_z_test (mu0 , 105, sigma , alpha =0.05 , power =0.80)
print(f"To detect mu' = 105 with 80% power: n = { required_n }")

# Plot power curves
fig , ax = plt.subplots(figsize=(10 , 6))
mu_alternatives = np.linspace(100 , 115, 100)

for n in [25, 50, 100]:
    powers = [ power_z_test (mu0 , mu , sigma , n, alpha , 'greater')
            for mu in mu_alternatives ]
    ax.plot( mu_alternatives , powers , label=f'n = {n}')

ax.axhline(y=0.80, color='r', linestyle ='--', label='80% power')
ax.axhline(y=alpha, color='gray', linestyle =':', label=f'alpha ={alpha}')
ax.set_xlabel("True Mean (mu ')")
ax.set_ylabel('Power')
ax.set_title('Power Curves for Different Sample Sizes')
ax.legend()
ax.grid(True , alpha =0.3)
#plt.savefig('power_curves.png', dpi=150, bbox_inches='tight')
plt.show()

2.7 Complete Hypothesis Testing Example ​

python
import numpy as np
from scipy import stats
import matplotlib.pyplot as plt

def hypothesis_test_report(test_name, data=None, params=None, results=None):
    """ Generate a comprehensive hypothesis test report ."""
    print("=" * 60)
    print(f"HYPOTHESIS TEST REPORT: { test_name }")
    print("=" * 60)

    if data is not None:
        print(f"\nData Summary:")
        print(f" Sample size: {len(data)}")
        print(f" Sample mean: {np.mean(data):.4f}")
        print(f" Sample std: {np.std(data , ddof =1) :.4f}")

    if params:
        print(f"\nTest Parameters:")
        for key , value in params.items ():
            print(f" {key}: {value}")

    if results :
        print(f"\nResults:")
        for key, value in results .items ():
            if isinstance (value , float):
                print(f" {key }: {value :.4f}")
            else:
                print(f" {key }: {value}")
    print("=" * 60)

# Simulation : Testing if a communication channel meets SNR requirements
np. random.seed(42)

# Simulated SNR measurements (dB) - should be at least 10 dB
snr_measurements = np.array ([10.2 , 9.8, 10.5 , 11.1 , 9.5, 10.8 , 10.3 , 9.7,
       10.9 , 10.1 , 9.9, 10.4 , 10.6 , 9.6, 10.7 , 10.0])

# Hypothesis test: H0: mu >= 10, Ha: mu < 10
mu0 = 10.0
alpha = 0.05

# Perform one -sample t-test
t_stat , p_value_two = stats. ttest_1samp ( snr_measurements , mu0)
p_value = p_value_two / 2 if t_stat < 0 else 1 - p_value_two / 2

# Effect size
effect_size = (np.mean( snr_measurements ) - mu0) / np.std( snr_measurements , ddof =1)

# Decision
reject_h0 = p_value < alpha and t_stat < 0

# Generate report
params = {
    'Null Hypothesis ': 'mu >= 10 dB',
    'Alternative ': 'mu < 10 dB',
    'Significance Level ': alpha ,
    'Test Type ': 'One -sample t-test (lower -tailed )'
}

results = {
    't-statistic ': t_stat ,
    'p-value ': p_value ,
    "Cohen 's d": effect_size ,
    'Critical Value ': stats.t.ppf(alpha , len( snr_measurements ) -1),
    'Decision ': 'Reject H0' if reject_h0 else 'Fail to Reject H0'
}

hypothesis_test_report ("SNR Requirement Verification ",
    snr_measurements , params , results )

# Visualization
fig , axes = plt. subplots (1, 2, figsize =(12 , 4))

# Histogram with reference line
axes [0]. hist( snr_measurements , bins=8, edgecolor ='black', alpha =0.7)
axes [0]. axvline (x=mu0 , color='r', linestyle ='--', linewidth =2,
    label=f'Required : {mu0} dB')
axes [0]. axvline (x=np.mean( snr_measurements ), color='g', linestyle ='-',
    linewidth =2, label=f'Sample Mean: {np.mean( snr_measurements ):.2f} dB')
axes [0]. set_xlabel ('SNR (dB)')
axes [0]. set_ylabel ('Frequency ')
axes [0]. set_title ('Distribution of SNR Measurements ')
axes [0]. legend ()

# t-distribution with test statistic
x = np. linspace (-4, 4, 100)
df = len( snr_measurements ) - 1
y = stats.t.pdf(x, df)
axes [1]. plot(x, y, 'b-', linewidth =2)
axes [1]. fill_between (x[x <= stats.t.ppf(alpha , df)],

axes [1]. axvline (x=t_stat, color='g', linestyle ='--', linewidth=2, label=f't = {t_stat :.3f}')
axes [1]. set_xlabel ('t')

axes [1]. set_title (f't- Distribution (df = {df})')
axes [1]. legend ()

if reject_h0 :
    print("The SNR does NOT meet the 10 dB requirement.")

    print(" Insufficient evidence that SNR fails to meet the 10 dB requirement.")
