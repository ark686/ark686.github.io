THE MEMORYLESS PROPERTY:

The exponential distribution is the only continuous distribution with the memoryless property:

P(X > s + t | X > s) = P(X > t) for all s, t ≥ 0

EXPONENTIAL-POSSON RELATIONSHIP:

If events occur according to a Poisson process with rate λ (i.e., the number of events in time t is Poisson with mean λt), then the time between consecutive events follows an Exponential(λ) distribution.

5.3 The Normal (Gaussian) Distribution ​

The normal distribution is arguably the most important distribution in probability and statistics. It appears naturally in many contexts due to the Central Limit Theorem.

A continuous random variable X has a normal distribution (or Gaussian distribution) with parameters µ (mean) and σ^2 (variance), written X ∼ N (µ, σ^2), if its PDF is:

PROPERTIES:

If X ∼ N(µ, σ^2), then:

  • E[X] = µ
  • Var(X) = σ^2
  • The PDF is symmetric about µ
  • The mode (peak) and median both equal µ
  • About 68% of probability lies within [µ − σ, µ + σ]
  • About 95% of probability lies within [µ − 2σ, µ + 2σ]
  • About 99.7% of probability lies within [µ − 3σ, µ + 3σ]

7.1 Using scipy.stats for Continuous Distributions ​

python
import numpy as np
from scipy import stats
import matplotlib.pyplot as plt

# ============================================================
# Uniform Distribution
# ============================================================
# X ~ Uniform(0, 10)
a, b = 0, 10
uniform_rv = stats.uniform(loc=a, scale=b-a)

print("=== Uniform(0, 10) ===")
print(f"Mean: {uniform_rv.mean():.4f}") # 5.0
print(f"Variance: {uniform_rv.var():.4f}") # 8.3333
print(f"P(2 < X < 7): {uniform_rv.cdf(7) - uniform_rv.cdf(2):.4f}") # 0.5

# ============================================================
# Exponential Distribution
# ============================================================
# X ~ Exponential(lambda = 0.5), so mean = 2
lam = 0.5
# scipy uses scale = 1/lambda
exp_rv = stats.expon(scale=1/lam)

print("\n=== Exponential (0.5) ===")
print(f"Mean: {exp_rv.mean():.4f}") # 2.0
print(f"Variance: {exp_rv.var():.4f}") # 4.0
print(f"P(X > 3): {1 - exp_rv.cdf(3):.4f}") # 0.2231

# Verify memoryless property
s, t = 2, 1
# P(X > s+t | X > s) should equal P(X > t)
p_conditional = (1 - exp_rv.cdf(s+t)) / (1 - exp_rv.cdf(s))
p_marginal = 1 - exp_rv.cdf(t)
print(f"Memoryless: P(X>{s+t}|X>{s}) = {p_conditional:.4f}")
print(f" P(X>{t}) = {p_marginal:.4f}") # Same!

# ============================================================
# Normal Distribution
# ============================================================
# X ~ Normal(100, 15^2)
mu, sigma = 100, 15
norm_rv = stats.norm(loc=mu, scale=sigma)

print("\n=== Normal(100, 225) ===")
print(f"Mean: {norm_rv.mean():.4f}") # 100
print(f"Std Dev: {norm_rv.std():.4f}") # 15
print(f"P(X < 120): {norm_rv.cdf(120):.4f}") # 0.9088
print(f"P(85 < X < 115): {norm_rv.cdf(115) - norm_rv.cdf(85):.4f}")

# Find 95th percentile
q95 = norm_rv.ppf(0.95)
print(f"95th percentile: {q95:.4f}") # 124.67

# Probability within 2 standard deviations
p_2sigma = norm_rv.cdf(mu + 2*sigma) - norm_rv.cdf(mu - 2*sigma)
print(f"P(mu-2sigma < X < mu+2sigma): {p_2sigma:.4f}") # ~0.9545

7.2 Generating Random Samples ​

python
import numpy as np
from scipy import stats
import matplotlib.pyplot as plt

np.random.seed(42)

# Generate samples from Normal(0, 1)
n_samples = 10000
samples = stats.norm.rvs(loc=0, scale=1, size=n_samples)

# Create figure
fig, axes = plt.subplots(1, 2, figsize=(12, 4))

# Left: Histogram vs theoretical PDF
ax1 = axes[0]
ax1.hist(samples , bins=50, density=True , alpha=0.7,
color='steelblue', edgecolor='white', label='Histogram')
x = np.linspace(-4, 4, 200)
ax1.plot(x, stats.norm.pdf(x), 'r-', lw=2, label='Theoretical PDF')
ax1.set_xlabel('x')
ax1.set_ylabel('Density')
ax1.set_title('Normal(0,1): Histogram vs PDF')
ax1.legend()

# Right: Empirical CDF vs theoretical CDF
ax2 = axes[1]
sorted_samples = np.sort(samples)
empirical_cdf = np.arange(1, n_samples+1) / n_samples
ax2.plot(sorted_samples , empirical_cdf , 'b-', lw=1,
label='Empirical CDF')
ax2.plot(x, stats.norm.cdf(x), 'r--', lw=2, label='Theoretical CDF')
ax2.set_xlabel('x')
ax2.set_ylabel('F(x)')
ax2.set_title('Empirical vs Theoretical CDF')
ax2.legend()

plt.tight_layout()
plt.savefig('normal_distribution_comparison.png', dpi=150)
plt.show()

# Sample statistics
print(f"Sample mean: {np.mean(samples):.4f} (theoretical: 0)")
print(f"Sample std: {np.std(samples , ddof=1):.4f} (theoretical: 1)")

7.3 Monte Carlo Estimation of Probabilities ​

python
import numpy as np
from scipy import stats

np.random.seed(42)

# Example: Estimate P(X^2 + Y^2 < 1) where X, Y ~ Uniform(-1, 1)
# This is pi/4 (area of unit circle / area of square)

n_simulations = 100000

# Generate uniform samples
X = stats.uniform.rvs(loc=-1, scale=2, size=n_simulations)
Y = stats.uniform.rvs(loc=-1, scale=2, size=n_simulations)

# Count points inside unit circle
inside_circle = (X**2 + Y**2) < 1
probability_estimate = np.mean(inside_circle)

print("=== Monte Carlo Estimation ===")
print(f"Estimated P(X^2 + Y^2 < 1): {probability_estimate:.4f}")
print(f"Theoretical value (pi/4): {np.pi/4:.4f}")
print(f"Estimated pi: {4 * probability_estimate:.4f}")

# Example: Transform distribution
# If X ~ Uniform(0, 1), find distribution of Y = -ln(X)
# Y should be Exponential(1)

X_samples = stats.uniform.rvs(size=n_simulations)
Y_samples = -np.log(X_samples)

# Compare with Exponential(1)
theoretical_mean = 1.0
theoretical_var = 1.0
print(f"\n=== Transform: Y = -ln(X) ===")
print(f"Sample mean of Y: {np.mean(Y_samples):.4f} (exp mean: 1.0)")
print(f"Sample var of Y: {np.var(Y_samples):.4f} (exp var: 1.0)")

7.4 Visualizing Multiple Distributions ​

python
import numpy as np
from scipy import stats
import matplotlib.pyplot as plt

fig, axes = plt.subplots(2, 3, figsize=(14, 8))

# Row 1: PDFs
# Uniform distributions
ax = axes[0, 0]
for (a, b) in [(0, 1), (0, 2), (-1, 1)]:
    x = np.linspace(a-0.5, b+0.5, 300)
    y = stats.uniform(loc=a, scale=b-a).pdf(x)
    ax.plot(x, y, lw=2, label=f'Uniform({a},{b})')
ax.set_xlabel('x')
ax.set_ylabel('f(x)')
ax.set_title('Uniform PDFs')
ax.legend()
ax.set_ylim(0, 1.5)

# Exponential distributions
ax = axes[0, 1]
x = np.linspace(0, 5, 300)
for lam in [0.5, 1.0, 2.0]:
    y = stats.expon(scale=1/lam).pdf(x)
    ax.plot(x, y, lw=2, label=f'Exp({lam})')
ax.set_xlabel('x')
ax.set_ylabel('f(x)')
ax.set_title('Exponential PDFs')
ax.legend()

# Normal distributions
ax = axes[0, 2]
x = np.linspace(-6, 10, 300)
for (mu, sigma) in [(0, 1), (2, 1), (0, 2)]:
    y = stats.norm(loc=mu, scale=sigma).pdf(x)
    ax.plot(x, y, lw=2, label=f'N({mu},{sigma}^2)')
ax.set_xlabel('x')
ax.set_ylabel('f(x)')
ax.set_title('Normal PDFs')
ax.legend()

# Row 2: CDFs
# Uniform CDFs
ax = axes[1, 0]
for (a, b) in [(0, 1), (0, 2), (-1, 1)]:
    x = np.linspace(a-0.5, b+0.5, 300)
    y = stats.uniform(loc=a, scale=b-a).cdf(x)
    ax.plot(x, y, lw=2, label=f'Uniform({a},{b})')
ax.set_xlabel('x')
ax.set_ylabel('F(x)')
ax.set_title('Uniform CDFs')
ax.legend()

# Exponential CDFs
ax = axes[1, 1]
x = np.linspace(0, 5, 300)
for lam in [0.5, 1.0, 2.0]:
    y = stats.expon(scale=1/lam).cdf(x)
    ax.plot(x, y, lw=2, label=f'Exp({lam})')
ax.set_xlabel('x')
ax.set_ylabel('F(x)')
ax.set_title('Exponential CDFs')
ax.legend()

# Normal CDFs
ax = axes[1, 2]
x = np.linspace(-6, 10, 300)
for (mu, sigma) in [(0, 1), (2, 1), (0, 2)]:
    y = stats.norm(loc=mu, scale=sigma).cdf(x)
    ax.plot(x, y, lw=2, label=f'N({mu},{sigma}^2)')
ax.set_xlabel('x')
ax.set_ylabel('F(x)')
ax.set_title('Normal CDFs')
ax.legend()

plt.tight_layout()
plt.savefig('distribution_comparison.png', dpi=150)
plt.show()
