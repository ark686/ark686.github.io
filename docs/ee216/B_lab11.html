2. Introduction ​

Root-finding is one of the most fundamental problems in numerical analysis. While some equations can be solved analytically (e.g., the quadratic formula for second-degree polynomials), most practical engineering equations require numerical methods.

Root-finding appears throughout ECE applications:

  1. Circuit Analysis: Finding operating points where nonlinear device characteristics (diodes, transistors) satisfy Kirchhoff’s laws
  2. Control Systems: Determining system poles and zeros from transfer functions; finding gain values for desired closed-loop behavior
  3. Signal Processing: Computing filter coefficients; finding frequencies where magnitude response equals specified values
  4. Power Systems: Solving power flow equations; determining fault currents
  5. Communications: Finding bit error rates; optimizing modulation parameters
  6. Semiconductor Physics: Solving for carrier concentrations; finding threshold voltages

2.1 Comparison of Methods ​

Single-Equation Root-Finding:

MethodConvergenceCost/IterRequirements
BisectionLinear1 evalValid bracket
Newton-RaphsonQuadraticf and f′Good initial guess
SecantSuperlinear (1.618)1 evalTwo initial guesses
Fixed-PointLinear1 eval|g′(x∗)|<1

Systems of Equations:

MethodConvergenceCost/IterRequirements
JacobiLinearLowDiag. dominance
Gauss-SeidelLinear (faster)LowDiag. dominance
Newton-RaphsonQuadraticHigh (Jacobian)Good initial guess

Guidelines:

  • Guaranteed convergence needed ⇒ Bisection or Brent’s method
  • Fast convergence, derivative available ⇒ Newton-Raphson
  • No derivative available ⇒ Secant or Brent’s method
  • Large sparse linear systems ⇒ Gauss-Seidel
  • Small nonlinear systems ⇒ Multivariate Newton-Raphson
  • Polynomial roots ⇒ numpy.roots

3.1 Bisection Method Implementation ​

python
import numpy as np

def bisection(f, xl, xu, tol=1e-6, max_iter=100):
    """
    Find root of f(x)= 0 using bisection method.

    Parameters :
        f: function to find root of
        xl, xu: lower and upper bounds(must bracket root)
        tol: tolerance for convergence
        max_iter : maximum iterations

    Returns :
        xr: estimated root
        iterations : list of(iteration, xr, error) tuples
    """
    if f(xl) * f(xu) > 0:
        raise ValueError("f(xl) and f(xu) must have opposite signs")

    iterations= []
    xr_old = xl

    for i in range(max_iter):
        xr = (xl + xu) / 2 # Midpoint
        fr = f(xr)

        # Approximate relative error
        if xr != 0:
            ea = abs((xr - xr_old) / xr) * 100
        else:
            ea = abs(xr - xr_old) * 100

        iterations.append((i+1, xr, ea))

        if ea < tol or abs(fr) < 1e-15:
            break

        if f(xl) * fr < 0:
            xu = xr
        else:
            xl = xr

        xr_old = xr

    return xr, iterations

# Example : Find root of f(x) = x^3 - x - 2
f = lambda x: x**3 - x - 2
root, iters = bisection(f, 1, 2)
print(f"Root: {root:.10f}")
print(f"Iterations: {len(iters)}")

3.2 Newton-Raphson Implementation ​

python
import numpy as np

def newton_raphson(f, df, x0, tol=1e-10, max_iter=50):
    """
    Find root using Newton - Raphson method.

    Parameters :
        f: function
        df: derivative of f
        x0: initial guess
        tol: tolerance
        max_iter : maximum iterations

    Returns :
        x: estimated root
        iterations : list of(iteration, x, f(x), error) tuples
    """
    x = x0
    iterations = []

    for i in range(max_iter):
        fx = f(x)
        dfx = df(x)

        if abs(dfx) < 1e-15:
            raise ValueError("Derivative too small")

        x_new = x - fx / dfx

        if x_new != 0:
            ea = abs((x_new - x) / x_new) * 100
        else:
            ea = abs(x_new - x) * 100

        iterations.append((i+1, x_new, fx, ea))

        if ea < tol:
            break

        x = x_new

    return x, iterations

# Example : Find root of f(x) = e^(-x) - x
f = lambda x: np.exp(-x) - x
df = lambda x: -np.exp(-x) - 1

root, iters = newton_raphson(f, df, 0)
print(f"Root: {root:.15f}")
print("\nConvergence history:")
for i, x, fx, ea in iters:
    print(f"Iter {i}: x = {x:.12f}, error = {ea:.2e}%")

3.3 Jacobi Method for Linear Systems ​

python
import numpy as np

def jacobi(A, b, x0, tol=1e-8, max_iter=100):
    """
    Jacobi iteration for solving Ax = b.

    Parameters :
        A: coefficient matrix(n x n)
        b: right-hand side vector
        x0: initial guess
        tol: convergence tolerance
        max_iter : maximum iterations
    """
    n = len(b)
    x = x0.copy()

    for k in range(max_iter):
        x_new = np.zeros(n)
        for i in range(n):
            sigma = sum(A[i,j] * x[j] for j in range(n) if j != i)
            x_new[i] =(b[i] - sigma) / A[i,i]

        # Check convergence
        if np.max(np.abs(x_new - x)) < tol:
            print(f"Jacobi converged in {k+1} iterations")
            return x_new

        x = x_new

    print("Max iterations reached")
    return x

# Example: 4x4 diagonally dominant system
A = np.array([[10, -1, 2, 1],
    [-1, 12, 1, -2],
    [2, 1, 8, 1],
    [1, 2, -1, 9]], dtype=float)
b = np.array([6, 18, -4, 12], dtype=float)
x0 = np.zeros(4)

solution = jacobi(A, b, x0)
print("Solution:", solution)
print("Verification: Ax =", A @ solution)

3.4 Gauss-Seidel Method for Linear Systems ​

python
import numpy as np

def gauss_seidel(A, b, x0, tol=1e-8, max_iter=100):
    """
    Gauss-Seidel iteration for solving Ax = b.
    Uses updated values immediately .
    """
    n = len(b)
    x = x0.copy()

    for k in range(max_iter):
        x_old = x.copy()
        for i in range(n):
            sigma = sum(A[i,j] * x[j] for j in range(n) if j != i)
            x[i] = (b[i] - sigma) / A[i,i]

        if np.max(np.abs(x - x_old)) < tol:
            print(f"Gauss -Seidel converged in {k+1} iterations")
            return x

    print("Max iterations reached")
    return x

# Same example system
A = np.array([[10, -1, 2, 1],
    [-1, 12, 1, -2],
    [2, 1, 8, 1],
    [1, 2, -1, 9]], dtype=float)
b = np.array([6, 18, -4, 12], dtype=float)
x0 = np.zeros(4)

solution = gauss_seidel(A, b, x0)
print("Solution:", solution)

root_fsolve = fsolve(f, x0=1.5)[0]
print(f"fsolve: {root_fsolve:.15f}")

# Verify
print(f"\nVerification: f(root) = {f(root_brent):.2e}")

3.6 Convergence Comparison Plot ​

python
import numpy as np
import matplotlib.pyplot as plt

def compare_methods(f, df, xl, xu, x0, true_root):
    """ Compare convergence of different root-finding methods."""

    # Bisection
    bis_errors = []
    a, b = xl, xu
    for i in range(20):
        x =(a + b) / 2
        bis_errors.append(abs(x - true_root))
        if f(a) * f(x) < 0:
            b = x
        else:
            a = x

    # Newton - Raphson
    nr_errors = []
    x = x0
    for i in range(10):
        nr_errors.append(abs(x - true_root))
        if abs(df(x)) < 1e-15:
            break
        x = x - f(x) / df(x)

    # Plot
    plt.figure(figsize=(10, 6))
    plt.semilogy(bis_errors, 'bo-', label='Bisection', markersize=8)
    plt.semilogy(nr_errors, 'rs-', label='Newton-Raphson', markersize=8)
    plt.xlabel('Iteration', fontsize=12)
    plt.ylabel('Absolute Error', fontsize=12)
    plt.title('Convergence Comparison', fontsize=14)
    plt.legend(fontsize=11)
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    #plt.savefig('convergence_comparison.png', dpi=150)
    plt.show()

# Example
f = lambda x: x**3 - x - 2
df = lambda x: 3*x**2 - 1
true_root = 1.5213797068045676

compare_methods(f, df, 1, 2, 1.5, true_root)

3.7 Solving Nonlinear Systems ​

python
import numpy as np
from scipy. optimize import fsolve

def system(x):
    """
    System of nonlinear equations:
    f1: x1^2 + x1*x2 - 10 = 0
    f2: x2 + 3*x1*x2^2 - 57 = 0
    """
    x1, x2 = x
    f1 = x1 **2 + x1*x2 - 10
    f2 = x2 + 3*x1*x2 **2 - 57
    return [f1, f2]

def jacobian(x):
    """ Jacobian matrix of the system ."""
    x1, x2 = x
    J = np.array([
        [2* x1 + x2, x1],
        [3* x2**2, 1 + 6*x1*x2]
    ])
    return J

# Solve using fsolve
x0 = [1.5, 3.5]
solution = fsolve(system, x0, fprime=jacobian)
print(f" Solution: x1 = {solution [0]:.10f}, x2 = {solution [1]:.10f}")
print(f" Verification: f1 = {system(solution)[0]:.2e}, "
    f"f2 = {system(solution)[1]:.2e}")

# Manual Newton-Raphson for systems
def newton_system(F, J, x0, tol=1e-10, max_iter=20):
    """Newton-Raphson for systems of equations ."""
    x = np.array(x0, dtype=float)

    for i in range(max_iter):
        Fx = np.array(F(x))
        Jx = J(x)
        dx = np.linalg.solve(Jx, -Fx)
        x = x + dx

        if np.linalg.norm(dx) < tol:
            print(f" Converged in {i+1} iterations")
            break

    return x

solution_manual = newton_system(system, jacobian, [1.5, 3.5])
print(f"Manual N-R: x1 = {solution_manual[0]:.10f}, "
    f"x2 = {solution_manual[1]:.10f}")

3.8 Newton-Raphson for Three-Equation Nonlinear System ​

python
import numpy as np

def F(x):
    """System : x1 ^2+ x2 ^2+ x3 ^2 -9=0, x1*x2+x2*x3 -4=0, x1+x2+x3 -5=0"""
    x1, x2, x3 = x
    return np.array([
        x1 **2 + x2 **2 + x3 **2 - 9,
        x1*x2 + x2*x3 - 4,
        x1 + x2 + x3 - 5
    ])

def J(x):
    """ Jacobian matrix ."""
    x1, x2, x3 = x
    return np.array([
        [2*x1, 2*x2, 2*x3],
        [x2, x1+x3, x2],
        [1, 1, 1]
    ])

# Newton-Raphson iteration
x = np.array([1.0, 2.0, 2.0])
print(f"{'Iter':>4} {'x1':>10} {'x2':>10} {'x3':>10} {'||f||':>10}")
print("-" * 52)

for k in range(10):
    Fx = F(x)
    norm_f = np.linalg.norm(Fx)
    print(f"{k:4d} {x[0]:10.6f} {x[1]:10.6f} {x[2]:10.6f} {norm_f:10.2e}")

    if norm_f < 1e-12:
        print(f"Converged in {k} iterations.")
        break

    Jx = J(x)
    dx = np.linalg.solve(Jx, -Fx)
    x = x + dx

# Verify solution
print(f"\nSolution : x = [{x[0]:.6f}, {x[1]:.6f}, {x[2]:.6f}]")
print(f"Residual : f = [{F(x)[0]:.2e}, {F(x)[1]:.2e}, {F(x)[2]:.2e}]")

# Compare with scipy.optimize.fsolve
from scipy.optimize import fsolve
x_scipy = fsolve(F, [1.0, 2.0, 2.0], fprime=J)
print(f"fsolve : x = [{x_scipy[0]:.6f}, {x_scipy[1]:.6f}, {x_scipy[2]:.6f}]")

3.9 Fixed-Point Iteration for Nonlinear System ​

python
import numpy as np

def jacobi_nonlinear(x0, tol =1e-8, max_iter=50):
    """
    Jacobi-style fixed-point iteration for:
        12* x1 + x2^2 - 3*x3 = 92
        x1 + 8*x2 - x3 = -8
        2*x1 + x2^3 - 20* x3 = -15
    """
    x = np.array(x0, dtype=float)
    print(f"{'Iter':>4} {'x1':>10} {'x2':>10} {'x3':>10}")
    print("-" * 40)
    print(f"{0:4d} {x[0]:10.5f} {x[1]:10.5f} {x[2]:10.5f}")

    for k in range(1, max_iter + 1):
        x_new = np.array([
           (92 - x[1]**2 + 3*x[2]) / 12,
           (-8 - x[0] + x[2]) / 8,
           (2*x[0] + x[1]**3 + 15) / 20
        ])

        err = np.linalg.norm(x_new - x, ord=np.inf)
        x = x_new.copy()
        print(f"{k:4d} {x[0]:10.5f} {x[1]:10.5f} {x[2]:10.5f}")

        if err < tol:
            print(f"\nConverged in {k} iterations(max change: {err:.2e})")
            break

    return x

# Run Jacobi -style iteration
x_jac = jacobi_nonlinear([1.0, 1.0, 1.0])

# Verify solution
print(f"\nVerification:")
x1, x2, x3 = x_jac
print(f"12* x1 + x2^2 - 3*x3 = {12* x1 + x2 **2 - 3*x3:.6f}(= 92)")
print(f"x1 + 8*x2 - x3 = {x1 + 8*x2 - x3:.6f}(= -8)")
print(f"2*x1 + x2^3 - 20* x3 = {2* x1 + x2 **3 - 20* x3:.6f}(= -15)")

# Compare: Gauss -Seidel (use updated values immediately)
print("\n--- Gauss-Seidel comparison ---")
x = np.array([1.0, 1.0, 1.0])
for k in range(1, 8):
    x[0] =(92 - x[1]**2 + 3*x[2]) / 12
    x[1] =(-8 - x[0] + x[2]) / 8
    x[2] =(2*x[0] + x[1]**3 + 15) / 20
    print(f"Iter {k}: x =[{x[0]:.5f}, {x[1]:.5f}, {x[2]:.5f}]")
print("Gauss-Seidel converges faster (5 vs 12 iterations).")
