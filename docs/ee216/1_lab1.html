2. NumPy Performance ​

The core of NumPy is well-optimized C code. Enjoy the flexibility of Python with the speed of compiled code.

Demonstrate NumPy's performance compared to "native" Python:

python
# Python program to demonstrate performance difference
# between NumPy and native Python

import numpy as np
import timeit

# 1. Using native Python lists and loops (naive approach)
def naive_dist(p, q):
    square_distance = 0
    for p_i, q_i in zip(p, q):
        square_distance += (p_i - q_i) ** 2
    return square_distance ** 0.5

# 2. Using NumPy's vectorized operations
def simple_numpy_dist(p_np, q_np):
    # Operations are applied to entire arrays simultaneously in optimized C code
    return np.sum((p_np - q_np) ** 2) ** 0.5

# Setup large datasets
list_p = [i for i in range(1_000_000)]
list_q = [i + 2 for i in range(1_000_000)]
numpy_p = np.arange(1_000_000)
numpy_q = np.arange(1_000_000) + 2

# Benchmark the functions
print("Python list loop time:")
%timeit naive_dist(list_p, list_q) 

print("\nNumPy vectorized time:")
%timeit simple_numpy_dist(numpy_p, numpy_q)

python
# Slicing
import numpy as np

a = np.arange(10)
print(f"a = {a}")

# start:end (noninclusive):step
print(f"a[2:11:3] = {a[2:11:3]}")

# Last index is not included
print(f"a[:4] = {a[:4]}")

# All three slice components are not required
# By default, start is 0, end is the last and step is 1
print(f"a[1:3] = {a[1:3]}")
print(f"a[::2] = {a[::2]}")
print(f"a[3:] = {a[3:]}")
